## Basic API Usage

### Example 1: Detect Threats in Image

```python
import requests
import cv2
from pathlib import Path

# Read image

    files = {'file': f}

    print(f"  - Class: {threat['class']}")

```

## Real-time Monitoring

### Example 5: WebSocket Real-time Updates

```python
import asyncio
import websockets
import json

async def monitor_threats():
    """Monitor threats in real-time via WebSocket"""
    
    uri = "ws://localhost:8000/ws"
    
    async with websockets.connect(uri) as websocket:
        print("Connected to threat detection system")
        
        while True:
            # Send ping
            await websocket.send_json({
                'type': 'request_update',
                'timestamp': str(datetime.now())
            })
            
            # Receive updates
            try:
                message = await asyncio.wait_for(
                    websocket.recv(), 
                    timeout=2.0
                )
                data = json.loads(message)
                
                if data['type'] == 'threat_detection':
                    threats = data['threats']
                    print(f"\n⚠️ {len(threats)} threats detected!")
                    
                    for threat in threats:
                        print(f"  - {threat['class']}: "
                              f"{threat['threat_level']} "
                              f"({threat['confidence']:.1%})")
                
                elif data['type'] == 'crowd_analysis':
                    crowd = data['data']
                    print(f"\n👥 Crowd: {crowd['stats']['crowd_percentage']:.1f}%")
                
                elif data['type'] == 'evacuation_plan':
                    print(f"\n🚪 Evacuation routes updated")
                
            except asyncio.TimeoutError:
                pass
            except Exception as e:
                print(f"Error: {e}")
                break

# Run async monitor
asyncio.run(monitor_threats())
```

### Example 6: Video Stream Processing

```python
import cv2
import requests
from collections import deque

def process_video_stream(video_path):
    """Process video stream with threat detection"""
    
    cap = cv2.VideoCapture(video_path)
    threats_history = deque(maxlen=30)
    
    while True:
        ret, frame = cap.read()
        if not ret:
            break
        
        # Save frame temporarily
        cv2.imwrite('temp_frame.jpg', frame)
        
        # Detect threats
        with open('temp_frame.jpg', 'rb') as f:
            response = requests.post(
                'http://localhost:8000/detect/frame',
                files={'file': f}
            )
        
        detection = response.json()
        threats_history.append(detection['threats'])
        
        # Draw on frame
        for threat in detection['threats']:
            x1, y1, x2, y2 = threat['bbox']
            color = (0, 0, 255) if threat['threat_level'] == 'CRITICAL' else (0, 255, 0)
            
            cv2.rectangle(frame, (x1, y1), (x2, y2), color, 2)
            cv2.putText(frame, threat['class'], (x1, y1-10),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.5, color, 2)
        
        # Display
        cv2.imshow('Threat Detection', frame)
        
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break
    
    cap.release()
    cv2.destroyAllWindows()

# Example usage
process_video_stream('aerial_footage.mp4')
```

### Example 8: Custom Threat Classifier

```python
from threat_detection import AerialThreatDetector

class CustomThreatDetector(AerialThreatDetector):
    """Extended detector with custom threat classification"""
    
    def __init__(self, yolo_model_path):
        super().__init__(yolo_model_path)
        
        # Custom threat mapping for your domain
        self.threat_classes = {
            'military_drone': True,
            'quadcopter': True,
            'fighter_jet': True,
            'cargo_drone': False,  # Not a threat
            'civilian_aircraft': False,
            'bird': False,
            'cloud': False
        }
        
        # Custom threat parameters
        self.threat_parameters = {
            'military_drone': {
                'speed_threshold': 30,  # m/s
                'proximity_danger_zone': 500,  # meters
                'size_threshold': 1000  # pixels
            },
            'fighter_jet': {
                'speed_threshold': 100,
                'proximity_danger_zone': 2000,
                'size_threshold': 5000
            }
        }
    
    def classify_threat_severity(self, threat, altitude=None, 
                                proximity_to_crowd=None):
        """Override with custom logic"""
        threat_class = threat['class'].lower()
        
        # Get custom parameters
        params = self.threat_parameters.get(threat_class, {})
        
        # Custom scoring logic
        severity_score = 0.0
        
        # Speed-based scoring
        if altitude and altitude < 100:
            severity_score += 0.4
        
        # Size-based scoring
        area = threat['area']
        if area > params.get('size_threshold', 5000):
            severity_score += 0.3
        
        # Proximity scoring
        if proximity_to_crowd and proximity_to_crowd < params.get('proximity_danger_zone', 500):
            severity_score += 0.3
        
        # Confidence
        severity_score += threat['confidence'] * 0.1
        
        threat['severity_score'] = min(severity_score, 1.0)
        return threat
```

### Example 10: Email Notifications

```python
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart

class EmailAlerter:
    """Send email alerts"""
    
    def __init__(self, smtp_server, sender_email, sender_password):
        self.smtp_server = smtp_server
        self.sender_email = sender_email
        self.sender_password = sender_password
    
    def send_alert(self, recipient_email, threats, impact_zone):
        """Send email alert"""
        
        message = MIMEMultipart()
        message['From'] = self.sender_email
        message['To'] = recipient_email
        message['Subject'] = f'ALERT: {len(threats)} Aerial Threat(s) Detected'
        
        # Email body
        body = f"""
        Aerial Threat Detection Alert
        
        Number of Threats: {len(threats)}
        Alert Level: {'CRITICAL' if len(threats) > 0 else 'LOW'}
        
        Threat Details:
        """
        
        for threat in threats:
            body += f"""
            - Class: {threat['class']}
              Confidence: {threat['confidence']:.1%}
              Severity: {threat['threat_level']}
            """
        
        if impact_zone:
            body += f"""
            
        Impact Zone:
            Center: {impact_zone['impact_center']}
            Radius: {impact_zone['impact_radius']}m
            ETA: {impact_zone['time_to_impact']}s
            """
        
        message.attach(MIMEText(body, 'plain'))
        
        # Send email
        with smtplib.SMTP_SSL(self.smtp_server, 465) as server:
            server.login(self.sender_email, self.sender_password)
            server.send_message(message)

# Usage
alerter = EmailAlerter(
    'smtp.gmail.com',
    'your_email@gmail.com',
    'your_password'
)
alerter.send_alert('recipient@example.com', threats, impact_zone)
```

## Performance Optimization

### Example 11: Model Quantization for Faster Inference

```python
import torch
from ultralytics import YOLO

def quantize_model(model_path):
    """Quantize model for faster inference"""
    
    # Load model
    model = YOLO(model_path)
    
    # Export to quantized formats
    # INT8 quantization
    model.export(format='tflite', imgsz=640, optimize=True, dynamic=False)
    
    # For PyTorch
    quantized_model = torch.quantization.quantize_dynamic(
        model.model,
        {torch.nn.Linear},
        dtype=torch.qint8
    )
    
    return quantized_model

# Usage
quantized_model = quantize_model('yolov8x.pt')
```

### Example 12: Batch Processing

```python
from pathlib import Path
import cv2
import numpy as np

def batch_detect_threats(image_directory, batch_size=32):
    """Process multiple images efficiently"""
    
    detector = AerialThreatDetector('yolov8x.pt')
    
    # Load all images
    image_paths = list(Path(image_directory).glob('*.jpg'))
    images = [cv2.imread(str(p)) for p in image_paths]
    
    # Process in batches
    all_threats = []
    
    for i in range(0, len(images), batch_size):
        batch = images[i:i+batch_size]
        
        # Process batch
        for frame in batch:
            threats = detector.detect_threats(frame)
            all_threats.append(threats)
        
        print(f"Processed {min(i+batch_size, len(images))}/{len(images)}")
    
    return all_threats

# Usage
threats = batch_detect_threats('./data/images/', batch_size=16)
```
